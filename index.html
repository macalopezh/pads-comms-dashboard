Queries BigQuery Fuente de los datos
Query 1 — CVR (3d desde apertura) + Revenue (3d / 30d / 60d)
Tablas: GROWTH_COMM_TARGET_AUDIENCE · GROWTH_COMM_TARGET_AUDIENCE_CONTROL · FRAME_DAILY_CAMPAIGN_KPIS · TURING_SILVER_COMM_PERFO_OFF_SITE_RECURRING_DAILY
WITH
-- Primer open por seller en el track de retención (push) — todos los sites
first_open AS (
  SELECT N_cust_adv_id AS user_id, MIN(T_event_date) AS first_open_date
  FROM `meli-sbox.MADSCOREBI.TURING_SILVER_COMM_PERFO_OFF_SITE_RECURRING_DAILY`
  WHERE S_ambito = 'Retention' AND S_bu = 'MERCADO ADS' AND S_product = 'PADS'
    AND S_communication_name LIKE '%_RET_JRN_REC_%'
    AND T_event_date >= '2025-08-01' AND N_open > 0
  GROUP BY 1
),
treatment AS (
  SELECT user_id, SITE_ID, SUB_SCOPE AS journey_cause,
    CASE WHEN EXTRACT(DAYOFWEEK FROM SAFE_CAST(JSON_EXTRACT_SCALAR(PAYLOAD,"$['triggerDate']") AS DATE)) = 1
         THEN SAFE_CAST(JSON_EXTRACT_SCALAR(PAYLOAD,"$['triggerDate']") AS DATE) + 1
         WHEN EXTRACT(DAYOFWEEK FROM SAFE_CAST(JSON_EXTRACT_SCALAR(PAYLOAD,"$['triggerDate']") AS DATE)) = 7
         THEN SAFE_CAST(JSON_EXTRACT_SCALAR(PAYLOAD,"$['triggerDate']") AS DATE) + 2
         ELSE SAFE_CAST(JSON_EXTRACT_SCALAR(PAYLOAD,"$['triggerDate']") AS DATE) END AS trigger_date
  FROM `meli-bi-data.SBOX_ADVERTISINGDEMAND.GROWTH_COMM_TARGET_AUDIENCE`
  WHERE COMMUNICATION_TARGET = 'target_re_activation'
    AND SUB_SCOPE IN ('Otra Reputación','Sin Items Activos','Tuvo venta','No tuvo venta')
    AND SNAPSHOT_DATE >= '2025-08-01'
  QUALIFY ROW_NUMBER() OVER (PARTITION BY user_id ORDER BY SAFE_CAST(JSON_EXTRACT_SCALAR(PAYLOAD,"$['triggerDate']") AS DATE) ASC) = 1
),
control AS (
  SELECT user_id, SITE_ID, SUB_SCOPE AS journey_cause,
    CASE WHEN EXTRACT(DAYOFWEEK FROM SAFE_CAST(JSON_EXTRACT_SCALAR(PAYLOAD,"$['triggerDate']") AS DATE)) = 1
         THEN SAFE_CAST(JSON_EXTRACT_SCALAR(PAYLOAD,"$['triggerDate']") AS DATE) + 1
         WHEN EXTRACT(DAYOFWEEK FROM SAFE_CAST(JSON_EXTRACT_SCALAR(PAYLOAD,"$['triggerDate']") AS DATE)) = 7
         THEN SAFE_CAST(JSON_EXTRACT_SCALAR(PAYLOAD,"$['triggerDate']") AS DATE) + 2
         ELSE SAFE_CAST(JSON_EXTRACT_SCALAR(PAYLOAD,"$['triggerDate']") AS DATE) END AS trigger_date
  FROM `meli-bi-data.SBOX_ADVERTISINGDEMAND.GROWTH_COMM_TARGET_AUDIENCE_CONTROL`
  WHERE COMMUNICATION_TARGET = 'target_re_activation'
    AND SUB_SCOPE IN ('Otra Reputación','Sin Items Activos','Tuvo venta','No tuvo venta')
    AND SNAPSHOT_DATE >= '2025-08-01'
  QUALIFY ROW_NUMBER() OVER (PARTITION BY user_id ORDER BY SAFE_CAST(JSON_EXTRACT_SCALAR(PAYLOAD,"$['triggerDate']") AS DATE) ASC) = 1
),
audience AS (
  SELECT t.user_id, t.SITE_ID, 'treatment' AS group_type, t.journey_cause, t.trigger_date,
    o.first_open_date,
    COALESCE(o.first_open_date, t.trigger_date) AS cvr_base_date,
    (o.first_open_date IS NOT NULL) AS did_open
  FROM treatment t LEFT JOIN first_open o ON t.user_id = o.user_id
  UNION ALL
  SELECT c.user_id, c.SITE_ID, 'control', c.journey_cause, c.trigger_date,
    NULL, c.trigger_date, FALSE
  FROM control c
),
frame_daily AS (
  SELECT SELLER_ID, SIT_SITE_ID, FECHA, SUM(REVS_TOTAL_USD) AS revs_usd
  FROM `meli-sbox.MADSCOREBI.FRAME_DAILY_CAMPAIGN_KPIS`
  WHERE FECHA >= '2025-08-01'
  GROUP BY 1, 2, 3
),
post_metrics AS (
  SELECT a.user_id, a.group_type, a.journey_cause, a.did_open,
    DATE_TRUNC(a.trigger_date, MONTH) AS cohort_month,
    MAX(CASE WHEN f.FECHA BETWEEN a.cvr_base_date AND a.cvr_base_date + 3 AND f.revs_usd > 0 THEN 1 ELSE 0 END) AS cvr_converted_3d,
    SUM(CASE WHEN f.FECHA BETWEEN a.trigger_date AND a.trigger_date + 3   THEN COALESCE(f.revs_usd,0) ELSE 0 END) AS revs_3d,
    SUM(CASE WHEN f.FECHA BETWEEN a.trigger_date AND a.trigger_date + 30  THEN COALESCE(f.revs_usd,0) ELSE 0 END) AS revs_30d,
    SUM(CASE WHEN f.FECHA BETWEEN a.trigger_date AND a.trigger_date + 60  THEN COALESCE(f.revs_usd,0) ELSE 0 END) AS revs_60d
  FROM audience a
  LEFT JOIN frame_daily f ON a.user_id = f.SELLER_ID AND a.SITE_ID = f.SIT_SITE_ID
    AND f.FECHA BETWEEN a.trigger_date AND a.trigger_date + 60
  WHERE a.trigger_date <= CURRENT_DATE - 60
  GROUP BY 1,2,3,4,5
),
agg AS (
  SELECT cohort_month, group_type,
    COUNT(DISTINCT user_id) AS n_users,
    COUNTIF(did_open) AS n_opened,
    SAFE_DIVIDE(COUNTIF(did_open), COUNT(DISTINCT user_id)) AS open_rate,
    SAFE_DIVIDE(COUNTIF(cvr_converted_3d=1), COUNT(DISTINCT user_id)) AS cvr_3d,
    AVG(revs_3d) AS avg_revs_3d, AVG(revs_30d) AS avg_revs_30d, AVG(revs_60d) AS avg_revs_60d,
    SUM(revs_3d) AS total_revs_3d, SUM(revs_30d) AS total_revs_30d, SUM(revs_60d) AS total_revs_60d
  FROM post_metrics GROUP BY 1,2
)
SELECT
  FORMAT_DATE('%Y-%m', t.cohort_month)                                    AS mes,
  t.n_users AS n_trat, c.n_users AS n_ctrl, t.n_opened,
  ROUND(t.open_rate*100,1)                                                AS open_rate_pct,
  ROUND(t.cvr_3d*100,2) AS cvr_3d_trat, ROUND(c.cvr_3d*100,2) AS cvr_3d_ctrl,
  ROUND((t.cvr_3d - c.cvr_3d)*100,2)                                     AS cvr_3d_uplift_pp,
  ROUND(SAFE_DIVIDE(t.cvr_3d - c.cvr_3d, c.cvr_3d)*100,1)               AS cvr_3d_uplift_rel,
  ROUND((t.avg_revs_3d  - c.avg_revs_3d)  * t.n_users, 0)               AS rev_incr_3d,
  ROUND((t.avg_revs_30d - c.avg_revs_30d) * t.n_users, 0)               AS rev_incr_30d,
  ROUND((t.avg_revs_60d - c.avg_revs_60d) * t.n_users, 0)               AS rev_incr_60d,
  ROUND(SAFE_DIVIDE(t.avg_revs_60d - c.avg_revs_60d,
                    t.avg_revs_30d - c.avg_revs_30d), 2)                 AS ratio_60d_vs_30d
FROM agg t JOIN agg c ON t.cohort_month=c.cohort_month AND c.group_type='control'
WHERE t.group_type='treatment'
ORDER BY 1
Query 2 — Semanal (revenue 30d, sin apertura)
Tablas: GROWTH_COMM_TARGET_AUDIENCE · GROWTH_COMM_TARGET_AUDIENCE_CONTROL · FRAME_DAILY_CAMPAIGN_KPIS
WITH
treatment AS (
  SELECT user_id, SITE_ID, 'treatment' AS group_type, SUB_SCOPE AS journey_cause,
    CASE WHEN EXTRACT(DAYOFWEEK FROM SAFE_CAST(JSON_EXTRACT_SCALAR(PAYLOAD,"$['triggerDate']") AS DATE)) = 1
         THEN SAFE_CAST(JSON_EXTRACT_SCALAR(PAYLOAD,"$['triggerDate']") AS DATE) + 1
         WHEN EXTRACT(DAYOFWEEK FROM SAFE_CAST(JSON_EXTRACT_SCALAR(PAYLOAD,"$['triggerDate']") AS DATE)) = 7
         THEN SAFE_CAST(JSON_EXTRACT_SCALAR(PAYLOAD,"$['triggerDate']") AS DATE) + 2
         ELSE SAFE_CAST(JSON_EXTRACT_SCALAR(PAYLOAD,"$['triggerDate']") AS DATE) END AS trigger_date
  FROM `meli-bi-data.SBOX_ADVERTISINGDEMAND.GROWTH_COMM_TARGET_AUDIENCE`
  WHERE COMMUNICATION_TARGET = 'target_re_activation'
    AND SUB_SCOPE IN ('Otra Reputación','Sin Items Activos','Tuvo venta','No tuvo venta')
    AND SNAPSHOT_DATE >= '2025-08-01'
  QUALIFY ROW_NUMBER() OVER (PARTITION BY user_id ORDER BY SAFE_CAST(JSON_EXTRACT_SCALAR(PAYLOAD,"$['triggerDate']") AS DATE) ASC) = 1
),
control AS (
  SELECT user_id, SITE_ID, 'control' AS group_type, SUB_SCOPE AS journey_cause,
    CASE WHEN EXTRACT(DAYOFWEEK FROM SAFE_CAST(JSON_EXTRACT_SCALAR(PAYLOAD,"$['triggerDate']") AS DATE)) = 1
         THEN SAFE_CAST(JSON_EXTRACT_SCALAR(PAYLOAD,"$['triggerDate']") AS DATE) + 1
         WHEN EXTRACT(DAYOFWEEK FROM SAFE_CAST(JSON_EXTRACT_SCALAR(PAYLOAD,"$['triggerDate']") AS DATE)) = 7
         THEN SAFE_CAST(JSON_EXTRACT_SCALAR(PAYLOAD,"$['triggerDate']") AS DATE) + 2
         ELSE SAFE_CAST(JSON_EXTRACT_SCALAR(PAYLOAD,"$['triggerDate']") AS DATE) END AS trigger_date
  FROM `meli-bi-data.SBOX_ADVERTISINGDEMAND.GROWTH_COMM_TARGET_AUDIENCE_CONTROL`
  WHERE COMMUNICATION_TARGET = 'target_re_activation'
    AND SUB_SCOPE IN ('Otra Reputación','Sin Items Activos','Tuvo venta','No tuvo venta')
    AND SNAPSHOT_DATE >= '2025-08-01'
  QUALIFY ROW_NUMBER() OVER (PARTITION BY user_id ORDER BY SAFE_CAST(JSON_EXTRACT_SCALAR(PAYLOAD,"$['triggerDate']") AS DATE) ASC) = 1
),
audience AS (SELECT * FROM treatment UNION ALL SELECT * FROM control),
frame_daily AS (
  SELECT SELLER_ID, SIT_SITE_ID, FECHA, SUM(REVS_TOTAL_USD) AS revs_usd
  FROM `meli-sbox.MADSCOREBI.FRAME_DAILY_CAMPAIGN_KPIS`
  WHERE FECHA >= '2025-08-01'
  GROUP BY 1, 2, 3
),
post_metrics AS (
  SELECT a.user_id, a.group_type,
    DATE_TRUNC(a.trigger_date, WEEK(MONDAY)) AS cohort_week,
    MAX(CASE WHEN f.revs_usd > 0 THEN 1 ELSE 0 END) AS converted,
    COALESCE(SUM(f.revs_usd), 0) AS revs_30d
  FROM audience a
  LEFT JOIN frame_daily f ON a.user_id = f.SELLER_ID AND a.SITE_ID = f.SIT_SITE_ID
    AND f.FECHA BETWEEN a.trigger_date AND a.trigger_date + 30
  WHERE a.trigger_date <= CURRENT_DATE - 30
  GROUP BY 1,2,3
),
agg AS (
  SELECT cohort_week, group_type,
    COUNT(DISTINCT user_id) AS n_users,
    SAFE_DIVIDE(COUNTIF(converted=1), COUNT(DISTINCT user_id)) AS cvr,
    AVG(revs_30d) AS avg_revs
  FROM post_metrics GROUP BY 1,2
)
SELECT
  FORMAT_DATE('W%V/%g', t.cohort_week) AS semana, t.cohort_week,
  t.n_users AS n_trat, c.n_users AS n_ctrl,
  ROUND(t.cvr*100,2) AS cvr_trat, ROUND(c.cvr*100,2) AS cvr_ctrl,
  ROUND((t.cvr - c.cvr)*100,2) AS cvr_uplift_pp,
  ROUND(SAFE_DIVIDE(t.cvr - c.cvr, c.cvr)*100,1) AS cvr_uplift_rel,
  ROUND((t.avg_revs - c.avg_revs) * t.n_users, 0) AS revs_incremental
FROM agg t JOIN agg c ON t.cohort_week=c.cohort_week AND c.group_type='control'
WHERE t.group_type='treatment'
ORDER BY 2
